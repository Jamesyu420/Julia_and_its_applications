using Random, Distributions, HypothesisTests
function f_paraci(
    n::Int,     # 
    p::Float64, # 
    M::Int)     # 
    # WaldWilson
    # alpha
    ncov = [0 0 0;
            0 0 0]
    alphas = [0.10, 0.05, 0.01]
    for i = 1:M
        x::Int = rand(Binomial(n, p))
        if x==0 || x==n
            continue
        end
        testr = BinomialTest(x, n, 0.5)
        for j in eachindex(alphas)
            low,hi = confint(testr, alphas[j], method=:wald)
            if low < p < hi
                ncov[1,j] += 1
            end
            low,hi = confint(testr, alphas[j], method=:wilson)
            if low < p < hi
                ncov[2,j] += 1
            end
        end # for j
    end # for i
    cover = ncov ./ M
    println("n=", n, " p=", p)
    println(cover)
    cover
end

function testone(x::Vector{Float64}, n1=30, n2=30)
    ndays = length(x)
    t = 1
    earn = 0.0
    while t < ndays - (n1 + n2)
        # n1
        lo, hi = quantile(x[t:(t+n1-1)], [0.2, 0.8])
        # n2
        t += n1
        t1 = t
        p_buy = NaN
        p_sell = NaN
        for t1 in t:(t+n2-2)
            if x[t1] < lo
                p_buy = x[t1]
                break
            end
        end
        if !isnan(p_buy) # t1+1t+n2-1
            for t2=(t1+1):(t+n2-1)
                if x[t2] > hi
                    p_sell = x[t2]
                    earn += p_sell  - p_buy
                    t = t2 + 1 # 
                    break
                end
            end
            # n2
            if isnan(p_sell)
                p_sell = x[t+n2-1]
                earn += p_sell - p_buy
                t += n2
            end
        end
    end # while t
    return earn
end
